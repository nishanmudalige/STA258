\section{Suggested Exercises}\label{ssec.se6}\markright{\ref{ssec.se6}
\titleref{ssec.se6}}
\begin{enumerate}
\item Find the basis and dimension of the solution space to the
homogeneous system
\begin{eqnarray*}x_1+2x_2+3x_3+4x_4 &=&0\\
x_2+2x_3+x_4&=&0\\ x_1+4x_3+5x_5&=&0 \end{eqnarray*}
\item Let
\begin{enumerate}
\item $A=\left [\begin {array}{rrrrr} 1&1&6&3&1\\
2&5&3&12&6\\ 1&3&0&7&4 \end {array}\right ]$
\item  $A=\left [\begin {array}{rrrrr} -1&2&0&1&-4\\
2&-4&3&1&3\\ 1&-2&1&0&2\\ -3&6&-4&-1&-4 \end{array} \right ]$

\begin{enumerate}
\item Find a basis for the row space of $A$.
\item Find the rank of $A$.
\item Find a basis for the null space of $A$.
\item Find the nullity of $A$.
\end{enumerate}
\end{enumerate}
\item Find the general solution to the following systems
\begin{enumerate}
\item  \begin{eqnarray*} x_1+2x_2+x_3+ 2x_4+x_5&=&8\\ x_1+ x_3+
2x_4 &=&9\\ 2x_2+ x_4+x_5 &=& 7
\end{eqnarray*}
\item
\begin{eqnarray*} x_1+x_2+2x_3+3x_4+3x_5&=&2\\
x_2+2x_3+x_4&=&3\\ x_1+2x_2+4x_3+5x_4+3x_5&=&7\\ x_1-x_2-2x_3-x_4+4x_5&=&-8
\end{eqnarray*}
\item
\begin{eqnarray*} x_1+2x_2+x_3+x_5-x_6&=&1\\
3x_1+6x_2+6x_4+6x_6&=&12\\ x_1+2x_2+x_3-2x_5+x_6&=&2\\ 2x_1+4x_2+2x_3+2x_5-3x_6&=&-1
\end{eqnarray*}
\end{enumerate}
\item Find the least square solution to $A{\bf{x}} = {\bf{b}}$
when
$$
A = \begin{bmatrix} 1 & 2 \\ 2 & 1 \\ 1 & 1 \end{bmatrix},\quad
{\bf{b}} = \begin{bmatrix} 8\\7\\4\end{bmatrix}.
$$
\item Find the straight line that best fits the following sets of data points
$$
(1,\ 3.5), (2,\ 10.8), (3,\ 18.1), (4,\ 25.1).
$$
\end{enumerate}

\item
If the equation of the regression line is $y = a_0 + a_1x$, then we want the least square solution to be
\begin{align*}
3.5&= a_0 + a_1\\
10.8 &= a_0 + 2a_2\\
18.1&= a_0 + 3a_3\\
25.1&=a_0+4a_4
\end{align*}
that is,
\begin{align*}
\begin{bmatrix} 1 & 1\\1 & 2\\1& 3\\1 & 4 \end{bmatrix}
\begin{bmatrix} a_0 \\ a_1 \end{bmatrix} = \begin{bmatrix} 3.5 \\10.8\\18.1\\25.1\end{bmatrix}.
\end{align*}
The solution is
\begin{align*}
\hat{\bf{a}} &= \left( \begin{bmatrix} 1 & 1 & 1 & 1 \\ 1 & 2 & 3 & 4 \end{bmatrix}
\begin{bmatrix} 1 & 1\\1 & 2\\1& 3\\1 & 4\end{bmatrix}\right)^{-1}
\begin{bmatrix} 1 & 1 & 1 & 1 \\ 1 & 2 & 3 & 4 \end{bmatrix}
\begin{bmatrix} 3.5 \\ 10.8 \\ 18.1 \\ 25.1 \end{bmatrix}\\
&=\frac{1}{20} \left[ \begin{array}{rr} 30 & -10 \\-10 & 4 \end{array} \right]
\begin{bmatrix}1 & 1 & 1 & 1\\1 & 2 & 3 & 4 \end{bmatrix}
\begin{bmatrix} 3.5 \\ 10.8 \\ 18.1 \\ 25.1 \end{bmatrix} \\
&= \left[ \begin{array}{r}-3.65 \\ 7.21 \end{array} \right].
\end{align*}

\end{enumerate}
